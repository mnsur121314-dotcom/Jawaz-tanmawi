// استبدل جزء الـ fetchStudents في الكود القديم بهذا الكود المحدث async function fetchStudents() { try { const { data, error } = await supabase.from('students').select('*'); if (error) { console.error("خطأ من Supabase:", error); document.getElementById('students-list').innerHTML = `
  • حدث خطأ: ${error.message}
    • `; return; } if (data.length === 0) { document.getElementById('students-list').innerHTML = `
  • الجدول فارغ في قاعدة البيانات.
    • `; return; } document.getElementById('students-list').innerHTML = data.map(s => `
  • ${s.full_name || 'بدون اسم'} - ${s.points || 0} نقطة
    • ` ).join(''); } catch (e) { console.error("خطأ عام:", e); } }
